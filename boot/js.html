JS 완전 마스터 (자주 보는 순서)

첫 화면 돌아가기

Variables / 변수

JS에서 데이터를 저장하는 그릇. let, const, var 사용

let num = 10;     // 변경 가능
const pi = 3.14;  // 변경 불가
var oldVar = 20;  // 예전 방식
  • let → 블록 범위 변수
  • const → 상수, 값 변경 불가
  • var → 함수 범위 변수, 요즘은 잘 사용하지 않음

Data Types / 데이터 타입

JS의 기본 타입과 typeof 확인

let str = "Hello";  // 문자열
let num = 100;      // 숫자
let flag = true;    // 불리언
let nothing = null; // 널
let undef;          // undefined
let obj = {name:"Alice"}; // 객체
let arr = [1,2,3];       // 배열

console.log(typeof str); // "string"
console.log(typeof obj); // "object"

Operators / 연산자

산술, 비교, 논리, 할당, 증감, 삼항 연산자

let a = 10, b = 3;

console.log(a + b);   // 덧셈 13
console.log(a - b);   // 뺄셈 7
console.log(a * b);   // 곱셈 30
console.log(a / b);   // 나눗셈 3.333...
console.log(a % b);   // 나머지 1

console.log(a > b);   // true
console.log(a === b); // false

console.log(a > 5 && b < 5); // true
console.log(a > 5 || b > 5); // true

let c = a > b ? a : b; // 삼항 연산자

Conditionals / 조건문

if, else if, else, switch

let score = 85;

if(score >= 90){
  console.log("A");
} else if(score >= 80){
  console.log("B");
} else {
  console.log("C");
}

switch(score){
  case 100: console.log("Perfect!"); break;
  case 85: console.log("Good"); break;
  default: console.log("Try again");
}

Loops / 반복문

for, while, do...while, for...of, for...in

for(let i=0;i<5;i++){
  console.log(i);
}

let j = 0;
while(j<3){
  console.log(j);
  j++;
}

let k = 0;
do{
  console.log(k);
  k++;
}while(k<2);

let arr = [10,20,30];
for(let val of arr){
  console.log(val);
}

let obj = {a:1, b:2};
for(let key in obj){
  console.log(key, obj[key]);
}

Functions / 함수

선언식, 표현식, 화살표 함수, 기본값, rest, spread

// 선언식
function add(x,y){ return x+y; }

// 표현식
const sub = function(x,y){ return x-y; }

// 화살표
const mul = (x,y) => x*y;

// 기본값
function greet(name="Guest"){
  console.log("Hello " + name);
}

// rest
function sumAll(...nums){
  return nums.reduce((a,b)=>a+b,0);
}

// spread
let arr1 = [1,2];
let arr2 = [...arr1,3,4]; // [1,2,3,4]

String / 문자열

텍스트 데이터를 다루는 타입과 메서드

let str = "Hello";

// 길이
console.log(str.length);

// 검색
console.log(str.indexOf("e"));
console.log(str.includes("H"));

// 자르기
console.log(str.slice(1,4));
console.log(str.substring(1,4));

// 변환
console.log(str.toUpperCase());
console.log(str.toLowerCase());

// 치환
console.log(str.replace("H","J"));

// 결합
let str2 = " World";
console.log(str + str2);
console.log(`${str}${str2}`);

Array / 배열

여러 데이터를 순서대로 담는 구조와 함수

let arr = [1,2,3,4];

// 추가/삭제
arr.push(5);
arr.pop();
arr.unshift(0);
arr.shift();

// 탐색
arr.indexOf(3);
arr.includes(2);

// 반복/변환
arr.forEach(num => console.log(num));
let squares = arr.map(num => num*num);
let filtered = arr.filter(num => num>2);
let sum = arr.reduce((acc,num)=>acc+num,0);

// 조작/정렬
arr.reverse();
arr.sort();
arr.slice(1,3);
arr.splice(1,2);
arr.join(", ");

Object / 객체

키와 값으로 데이터를 담는 구조와 함수

let person = {
  name: "Alice",
  age: 25,
  greet() { console.log(`Hi, I'm ${this.name}`); }
};

// 접근
console.log(person.name);
console.log(person["age"]);

// 객체 함수
Object.keys(person);
Object.values(person);
Object.entries(person);
Object.assign(person,{job:"Developer"});
Object.hasOwn(person,"age");

Reference Type / 참조 타입

객체, 배열, 함수 등 참조로 전달되는 타입

let arr1 = [1,2,3];
let arr2 = arr1; // 같은 참조
arr2.push(4);
console.log(arr1); // [1,2,3,4]

// 깊은 복사
let arr3 = structuredClone(arr1);
arr3.push(5);
console.log(arr1); // [1,2,3,4]
console.log(arr3); // [1,2,3,4,5]

DOM / 이벤트

문서 선택, 속성/값 변경, 이벤트 처리

const el = document.getElementById("myId");
const el2 = document.querySelector(".myClass");
