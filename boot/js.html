JavaScript

첫 화면 돌아가기

Variables / 변수

값을 저장할 수 있는 공간. var, let, const 사용.

var x = 10;     // 기존 방식, 함수 스코프
let y = 20;     // 블록 스코프
const z = 30;   // 상수, 재할당 불가

Data Types / 자료형

JS 기본 자료형

  • String → "문자열"
  • Number → 123, 3.14
  • Boolean → true / false
  • Null → 값 없음
  • Undefined → 값이 정의되지 않음
  • Object → { key: value }
  • Array → [1,2,3]

String / 문자열

텍스트를 다루는 도구. 길이 확인, 자르기, 검색, 변환, 결합 등

// 문자열 예제
let str = "Hello, JavaScript!";

// 길이
console.log(str.length); // 18

// 포함 여부
console.log(str.includes("Java"));   // true
console.log(str.includes("Python")); // false

// 시작/끝 확인
console.log(str.startsWith("Hello")); // true
console.log(str.endsWith("!"));       // true

// 자르기
console.log(str.slice(0,5));    // "Hello"
console.log(str.slice(-10));    // "JavaScript!"
console.log(str.substring(7,17));// "JavaScript"
console.log(str.substr(7,10));  // "JavaScript"

// 변환
console.log(str.toUpperCase()); // "HELLO, JAVASCRIPT!"
console.log(str.toLowerCase()); // "hello, javascript!"

// 공백 제거
let spaceStr = "   Hi there!   ";
console.log(spaceStr.trim());     // "Hi there!"
console.log(spaceStr.trimStart());// "Hi there!   "
console.log(spaceStr.trimEnd());  // "   Hi there!"

// 결합
let greet = "Hello" + " " + "World";
console.log(greet); // "Hello World"

let name = "Alice";
console.log(`Hi, ${name}! Welcome to JS.`);

// 검색
console.log(str.indexOf("Java"));  // 7
console.log(str.lastIndexOf("o"));// 4
console.log(str.search("Java"));  // 7

// 치환
console.log(str.replace("JavaScript","JS"));  // "Hello, JS!"
console.log(str.replaceAll("o","0"));         // "Hell0, JavaScript!"

// 분리/합치기
console.log(str.split(",")); // ["Hello", " JavaScript!"]
console.log(["Hello","JS"].join("-")); // "Hello-JS"

// 반복
console.log("Ha".repeat(3)); // "HaHaHa"

// 문자 접근
console.log(str.charAt(0)); // "H"
console.log(str[0]);         // "H"

// 패턴 매칭 (정규식)
console.log(str.match(/Java/)); // ["Java"]
console.log(str.search(/Java/)); // 7

// 비교
console.log("apple" > "banana"); // false
console.log("apple" === "apple"); // true

Array / 배열

여러 데이터를 순서대로 저장하고 다룰 수 있음

let arr = [1,2,3,4,5];

// 접근
console.log(arr[0]); // 1

// 길이
console.log(arr.length); // 5

// 추가
arr.push(6);      // 끝에 추가
arr.unshift(0);   // 앞에 추가

// 삭제
arr.pop();        // 끝 제거
arr.shift();      // 앞 제거

// 검색
console.log(arr.indexOf(3)); // 3 위치
console.log(arr.includes(2)); // true

// 반복
arr.forEach((v,i) => console.log(i, v));

// 변형
console.log(arr.map(x => x*2)); // [0,2,4,6,8,10]
console.log(arr.filter(x => x>2)); // [3,4,5]

// 정렬
console.log(arr.sort()); // 기본 문자열 순서
console.log(arr.reverse()); // 역순

Object / 객체

키와 값으로 데이터를 묶어서 저장

let obj = { name: "Alice", age: 23 };

// 접근
console.log(obj.name); // "Alice"
console.log(obj["age"]); // 23

// 추가/수정
obj.city = "Seoul";
obj.age = 24;

// 삭제
delete obj.city;

// 반복
for(let key in obj){
    console.log(key, obj[key]);
}

// 축약 표현
let x = 10, y = 20;
let point = { x, y }; // { x: 10, y: 20 }

// 메서드
let user = {
    name: "Bob",
    greet() { console.log("Hi, "+this.name); }
};
user.greet(); // "Hi, Bob"

Functions / 함수

코드 블록을 재사용 가능하게 만드는 도구

// 선언식
function add(a,b){
    return a+b;
}

// 표현식
let sub = function(a,b){ return a-b; }

// 화살표 함수
let mul = (a,b) => a*b;

// 호출
console.log(add(2,3)); // 5
console.log(sub(5,2)); // 3
console.log(mul(3,4)); // 12
