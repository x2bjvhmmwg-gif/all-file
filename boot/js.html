JavaScript

첫 화면 돌아가기

문자열 / String

길이 / length

let str = "Hello, JavaScript!";
// "Hello, JavaScript!"는 18글자
console.log(str.length); 
// 18 → 문자열에 포함된 글자 수를 반환

포함 여부 / includes

console.log(str.includes("Java"));   
// true → "Java"가 문자열에 포함됨
console.log(str.includes("Python")); 
// false → "Python"은 문자열에 없음

시작/끝 확인 / startsWith, endsWith

console.log(str.startsWith("Hello")); 
// true → 문자열이 "Hello"로 시작
console.log(str.endsWith("!"));       
// true → 문자열이 "!"로 끝남

문자열 자르기 / slice, substring, substr

console.log(str.slice(0,5));      
// "Hello" → 0부터 5 전까지 추출
console.log(str.substring(7,17));  
// "JavaScript" → 7부터 17 전까지 추출
console.log(str.substr(7,10)); 
// "JavaScript" → 7부터 10글자 추출 (사용 권장 안함)

변환 / toUpperCase, toLowerCase

console.log(str.toUpperCase()); 
// "HELLO, JAVASCRIPT!" → 모두 대문자로 변환
console.log(str.toLowerCase()); 
// "hello, javascript!" → 모두 소문자로 변환

공백 제거 / trim, trimStart, trimEnd

let spaceStr = "   Hi there!   ";
console.log(spaceStr.trim());     
// "Hi there!" → 앞뒤 공백 제거
console.log(spaceStr.trimStart()); 
// "Hi there!   " → 앞 공백만 제거
console.log(spaceStr.trimEnd());   
// "   Hi there!" → 뒤 공백만 제거

결합 / +, 템플릿 리터럴, concat

let greet = "Hello" + " " + "World";
console.log(greet); 
// "Hello World" → 문자열 결합

let name = "Alice";
console.log(`Hi, ${name}! Welcome to JS.`); 
// "Hi, Alice! Welcome to JS." → 템플릿 리터럴

console.log("Hello".concat(" ", "JS"));
// "Hello JS" → concat 메서드로 결합

검색 / indexOf, lastIndexOf, search

console.log(str.indexOf("Java"));  
// 7 → "Java" 시작 인덱스
console.log(str.lastIndexOf("o"));
// 4 → 마지막 "o" 위치
console.log(str.search("Java"));  
// 7 → 패턴 검색

치환 / replace, replaceAll

console.log(str.replace("JavaScript","JS"));  
// "Hello, JS!" → 첫 번째 "JavaScript"를 "JS"로 변경
console.log(str.replaceAll("o","0"));         
// "Hell0, JavaScript!" → 모든 "o"를 "0"으로 변경

분리/합치기 / split, join

console.log(str.split(",")); 
// ["Hello"," JavaScript!"] → 쉼표 기준으로 분리
console.log(["Hello","JS"].join("-")); 
// "Hello-JS" → 배열 요소를 "-"로 합침

반복 / repeat

console.log("Ha".repeat(3)); 
// "HaHaHa" → 문자열 반복

문자 접근 / charAt, []

console.log(str.charAt(0)); 
// "H" → 0번 인덱스 문자
console.log(str[0]);         
// "H" → 동일하게 접근 가능

패턴 매칭 / match, matchAll

console.log(str.match(/Java/)); 
// ["Java"] → 정규식과 매칭
let matches = str.matchAll(/a/g);
console.log([...matches].map(m=>m[0]));
// ["a","a"] → 모두 추출

문자 길이 맞추기 / padStart, padEnd

let str2 = "5";
console.log(str2.padStart(3,"0")); 
// "005" → 길이 3, 부족한 앞을 0으로 채움
console.log(str2.padEnd(4,"*"));   
// "5***" → 길이 4, 부족한 뒤를 *로 채움

문자열 비교 / >, <, ==, ===, localeCompare

console.log("apple" > "banana"); 
// false → "apple"은 "banana"보다 사전 순으로 뒤
console.log("apple" == "apple"); 
// true → 값이 같음
console.log("apple" === "apple"); 
// true → 값과 타입 모두 같음
console.log("apple".localeCompare("banana")); 
// -1 → apple이 앞이면 -1, 같으면 0, 뒤면 1

배열 / Array

생성 / [], Array()

let arr = [1,2,3,4];
let arr2 = new Array(5); 
console.log(arr); // [1,2,3,4]
console.log(arr2); // [empty ×5] → 길이만 5인 배열

요소 추가/삭제 / push, pop, unshift, shift

arr.push(5); 
// [1,2,3,4,5] → 뒤에 추가
arr.pop();  
// [1,2,3,4] → 뒤 제거
arr.unshift(0); 
// [0,1,2,3,4] → 앞에 추가
arr.shift();   
// [1,2,3,4] → 앞 제거

검색 / indexOf, includes

console.log(arr.indexOf(3)); 
// 2 → 값 3 위치
console.log(arr.includes(5)); 
// false → 배열에 5 없음

분리/합치기 / slice, splice, concat

console.log(arr.slice(1,3)); 
// [2,3] → 1~2번 인덱스 추출
arr.splice(1,2); 
// [2,3] → 삭제 후 반환, arr=[1,4]
console.log(arr.concat([5,6])); 
// [1,4,5,6] → 배열 합침

반복 / forEach, map, filter

let numbers = [1,2,3,4];
numbers.forEach(n=>console.log(n*2));
// 2,4,6,8 → 각 요소 *2 출력

let squares = numbers.map(n=>n*n);
console.log(squares); 
// [1,4,9,16] → 새로운 배열 반환

let evens = numbers.filter(n=>n%2===0);
console.log(evens); 
// [2,4] → 조건 만족 요소만 반환

검색 / find, findIndex

console.log(numbers.find(n=>n>2));   
// 3 → 조건 만족 첫 요소
console.log(numbers.findIndex(n=>n>2)); 
// 2 → 조건 만족 첫 요소 인덱스

기타 / join, reverse, sort, includes

console.log(numbers.join("-")); 
// "1-2-3-4"
console.log(numbers.reverse()); 
// [4,3,2,1]
console.log(numbers.sort((a,b)=>a-b)); 
// [1,2,3,4] → 오름차순

객체 / Object

생성 / {}, new Object()

let obj = {name:"Alice", age:25};
let obj2 = new Object();
obj2.city="Seoul";
console.log(obj); // {name:"Alice", age:25}
console.log(obj2); // {city:"Seoul"}

접근 / dot notation, bracket notation

console.log(obj.name); 
// "Alice"
console.log(obj["age"]); 
// 25

추가/수정 / dot, bracket

obj.gender = "female"; 
obj["age"] = 26;
console.log(obj); 
// {name:"Alice", age:26, gender:"female"}

삭제 / delete

delete obj.gender;
console.log(obj); 
// {name:"Alice", age:26}

키/값 / Object.keys, Object.values, Object.entries

console.log(Object.keys(obj)); 
// ["name","age"]
console.log(Object.values(obj)); 
// ["Alice",26]
console.log(Object.entries(obj)); 
// [["name","Alice"],["age",26]]

병합 / Object.assign

let obj3 = {country:"KR"};
let merged = Object.assign({}, obj, obj3);
console.log(merged); 
// {name:"Alice", age:26, country:"KR"}

비구조화 / Destructuring

let {name, age} = obj;
console.log(name, age); 
// "Alice", 26

함수 / Function

선언 / function, arrow function

function sum(a,b){ return a+b; }
console.log(sum(3,4)); 
// 7

const multiply = (x,y) => x*y;
console.log(multiply(2,5)); 
// 10

기본 매개변수 / default parameter

function greet(name="Guest"){ return `Hello ${name}`; }
console.log(greet()); 
// "Hello Guest"
console.log(greet("Alice")); 
// "Hello Alice"

나머지 매개변수 / rest parameter

function sumAll(...nums){ return nums.reduce((a,b)=>a+b,0); }
console.log(sumAll(1,2,3,4)); 
// 10

즉시 실행 함수 / IIFE

(function(){ console.log("즉시 실행"); })();
// "즉시 실행"

참조 타입 / Reference Types

배열/객체 참조

let a = [1,2];
let b = a;
b.push(3);
console.log(a); 
// [1,2,3] → a와 b는 같은 참조

let objA = {x:1};
let objB = objA;
objB.y=2;
console.log(objA); 
// {x:1, y:2} → 같은 참조

복사 / spread, Object.assign

let arrCopy = [...a];
arrCopy.push(4);
console.log(a); 
// [1,2,3] → 원본 유지

let objCopy = Object.assign({}, objA);
objCopy.z=3;
console.log(objA); 
// {x:1, y:2} → 원본 유지
